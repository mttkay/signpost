15  package oauth.signpost.jetty;
16  
17  import java.io.ByteArrayInputStream;
18  import java.io.IOException;
19  import java.io.InputStream;
20  import java.util.HashMap;
21  import java.util.Iterator;
22  import java.util.Map;
23  
24  import oauth.signpost.http.HttpRequest;
25  
26  import org.mortbay.jetty.HttpFields;
27  import org.mortbay.jetty.HttpFields.Field;
28  import org.mortbay.jetty.client.HttpExchange;
29  
30  public class HttpRequestAdapter implements HttpRequest {
31  
32      private HttpExchange request;
33  
34      private String requestUrl;
35  
36      public HttpRequestAdapter(HttpExchange request) {
37          this.request = request;
38          buildRequestUrl();
39      }
40  
41      public String getContentType() {
42          HttpFields fields = request.getRequestFields();
43          return fields.getStringField("Content-Type");
44      }
45  
46      public InputStream getMessagePayload() throws IOException {
47          return new ByteArrayInputStream(request.getRequestContent().array());
48      }
49  
50      public String getMethod() {
51          return request.getMethod();
52      }
53  
54      public String getRequestUrl() {
55          return requestUrl;
56      }
57  
58      public void setRequestUrl(String url) {
59          throw new RuntimeException(new UnsupportedOperationException());
60      }
61  
62      public void setHeader(String name, String value) {
63          request.setRequestHeader(name, value);
64      }
65  
66      public String getHeader(String name) {
67          HttpFields fields = request.getRequestFields();
68          return fields.getStringField(name);
69      }
70  
71      @SuppressWarnings("unchecked")
72      public Map<String, String> getAllHeaders() {
73          HttpFields fields = request.getRequestFields();
74          Iterator iter = fields.getFields();
75          HashMap<String, String> headers = new HashMap<String, String>();
76          while (iter.hasNext()) {
77              Field field = (Field) iter.next();
78              headers.put(field.getName(), field.getValue());
79          }
80  
81          return headers;
82      }
83  
84      // Jetty has some very weird mechanism for handling URLs... we have to
85      // reconstruct it here.
86      private void buildRequestUrl() {
87          StringBuilder sb = new StringBuilder();
88          sb.append(request.getScheme() + "://");
89          sb.append(request.getAddress().toString().replaceAll("://d+", ""));
90          if (request.getURI() != null) {
91              // the "URI" in Jetty is actually the path... WTF?!
92              sb.append(request.getURI());
93          }
94          this.requestUrl = sb.toString();
95      }
96  }
