15  package oauth.signpost.commonshttp;
16  
17  import oauth.signpost.AbstractOAuthConsumer;
18  import oauth.signpost.http.HttpRequest;
19  
20  /***
21   * Supports signing HTTP requests of type {@link org.apache.http.HttpRequest}.
22   * 
23   * @author Matthias Kaeppler
24   */
25  public class CommonsHttpOAuthConsumer extends AbstractOAuthConsumer {
26  
27      private static final long serialVersionUID = 1L;
28  
29      public CommonsHttpOAuthConsumer(String consumerKey, String consumerSecret) {
30          super(consumerKey, consumerSecret);
31      }
32  
33      @Override
34      protected HttpRequest wrap(Object request) {
35          if (!(request instanceof org.apache.http.HttpRequest)) {
36              throw new IllegalArgumentException(
37                      "This consumer expects requests of type "
38                              + org.apache.http.HttpRequest.class.getCanonicalName());
39          }
40  
41          return new HttpRequestAdapter((org.apache.http.client.methods.HttpUriRequest) request);
42      }
43  
44  }
