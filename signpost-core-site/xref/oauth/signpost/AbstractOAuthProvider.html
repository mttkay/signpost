48      public AbstractOAuthProvider(String requestTokenEndpointUrl, String accessTokenEndpointUrl,

57      public String retrieveRequestToken(OAuthConsumer consumer, String callbackUrl)
58              throws OAuthMessageSignerException, OAuthNotAuthorizedException,
59              OAuthExpectationFailedException, OAuthCommunicationException {
60  
61          // invalidate current credentials, if any
62          consumer.setTokenWithSecret(null, null);
63  
64          // 1.0a expects the callback to be sent while getting the request token.
65          // 1.0 service providers would simply ignore this parameter.
66          retrieveToken(consumer, OAuth.addQueryParameters(requestTokenEndpointUrl,
67              OAuth.OAUTH_CALLBACK, callbackUrl));
68  
69          String callbackConfirmed = responseParameters.get(OAuth.OAUTH_CALLBACK_CONFIRMED);
70          responseParameters.remove(OAuth.OAUTH_CALLBACK_CONFIRMED);
71          isOAuth10a = Boolean.TRUE.toString().equals(callbackConfirmed);
72  
73          // 1.0 service providers expect the callback as part of the auth URL,
74          // Do not send when 1.0a.
75          if (isOAuth10a) {
76              return OAuth.addQueryParameters(authorizationWebsiteUrl, OAuth.OAUTH_TOKEN, consumer
77                  .getToken());
78          } else {
79              return OAuth.addQueryParameters(authorizationWebsiteUrl, OAuth.OAUTH_TOKEN, consumer
80                  .getToken(), OAuth.OAUTH_CALLBACK, callbackUrl);
81          }
82      }
83  
84      public void retrieveAccessToken(OAuthConsumer consumer, String oauthVerifier)
85              throws OAuthMessageSignerException, OAuthNotAuthorizedException,
86              OAuthExpectationFailedException, OAuthCommunicationException {
87  
88          if (consumer.getToken() == null || consumer.getTokenSecret() == null) {
89              throw new OAuthExpectationFailedException(
90                  "Authorized request token or token secret not set. "
91                          + "Did you retrieve an authorized request token before?");
92          }
93  
94          String endpointUrl = isOAuth10a && oauthVerifier != null ? OAuth.addQueryParameters(
95              accessTokenEndpointUrl, OAuth.OAUTH_VERIFIER, oauthVerifier) : accessTokenEndpointUrl;
96  
97          retrieveToken(consumer, endpointUrl);
98      }

100     /***
101      * <p>
102      * Implemented by subclasses. The responsibility of this method is to
103      * contact the service provider at the given endpoint URL and fetch a
104      * request or access token. What kind of token is retrieved solely depends
105      * on the URL being used.
106      * </p>
107      * <p>
108      * Correct implementations of this method must guarantee the following
109      * post-conditions:
110      * <ul>
111      * <li>the {@link OAuthConsumer} passed to this method must have a valid
112      * {@link OAuth#OAUTH_TOKEN} and {@link OAuth#OAUTH_TOKEN_SECRET} set by
113      * calling {@link OAuthConsumer#setTokenWithSecret(String, String)}</li>
114      * <li>{@link #getResponseParameters()} must return the set of query
115      * parameters served by the service provider in the token response, with all
116      * OAuth specific parameters being removed</li>
117      * </ul>
118      * </p>
119      * 
120      * @param consumer
121      *        the {@link OAuthConsumer} that should be used to sign the request
122      * @param endpointUrl
123      *        the URL at which the service provider serves the OAuth token that
124      *        is to be fetched
125      * @throws OAuthMessageSignerException
126      *         if signing the token request fails
127      * @throws OAuthCommunicationException
128      *         if a network communication error occurs
129      * @throws OAuthNotAuthorizedException
130      *         if the server replies 401 - Unauthorized
131      * @throws OAuthExpectationFailedException
132      *         if an expectation has failed, e.g. because the server didn't
133      *         reply in the expected format
134      */
135     protected abstract void retrieveToken(OAuthConsumer consumer, String endpointUrl)
136 			throws OAuthMessageSignerException, OAuthCommunicationException,
137 			OAuthNotAuthorizedException, OAuthExpectationFailedException;
